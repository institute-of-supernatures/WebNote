## Vue Json 引入混淆方案
```js
// 通用的 JSON 数据混入工厂函数
export function createJsonDataMixin (dataFilePath, dataName = 'tableJson') {
  if (!dataFilePath) throw new Error('dataFilePath is required')
  const clonedFn = (data) => JSON.parse(JSON.stringify(data))
  const rowData = clonedFn(require(`${dataFilePath}`))

  // 冻结数据，使其不可变
  const frozenData = Object.freeze(rowData)

  return {
    data () {
      const data = {}
      data[dataName] = frozenData
      return data
    }
  }
}
```
