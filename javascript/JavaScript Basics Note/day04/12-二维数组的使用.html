二维数组的使用

  • 一维数组：数组中的元素是基本数据类型

  • 二维数组：数组中的每一个元素是一个数组

  • 三维数组：数组中的每一个元素是一个二维数组


 <script>
    /**
     * 二维数组：数组中的每一个元素是一个数组
     * 取值：数组名[行][列]     二维数组：可以当作一个多行多列的表格
     * 遍历：二维数组的遍历使用for循环嵌套
     *      * 一维数组的遍历使用一个for循环
     */

    //一维数组： 一个学校有3个班，每个班有3个人，那么学校需要把所有班级的成绩都保存起来
     var oneScores = [99, 98, 75];
    var twoScores = [93, 91, 86];
    var threeScores = [90, 0, 0];
    //二维数组：使用一个变量保存全校所有班级的学生成绩
    var school = [ [99, 98, 75],[93, 91, 86],[90, 0, 0]];//二维数组中的每一个元素都是一个数组

    //取值：数组名[行][列]    行决定第几个元素下标，列决定该元素中的子元素下标
    var zhangsan = school[1][2];//86
    var lisi = school[0][1];//98
    console.log ( zhangsan, lisi );

    //遍历：
    for (var i = 0; i < school.length; i++) {//第一层循环遍历行

        for (var j = 0; j < school[i].length; j++) {//第二层循环遍历列

            console.log(school[i][j]);//获取二维数组中的每一个数据
        }
    }

</script>
